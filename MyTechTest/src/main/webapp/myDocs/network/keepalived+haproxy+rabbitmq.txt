chmod +x /etc/init.d/haproxy
mkdir /etc/haproxy

/etc/haproxy/haproxy.cfg
global
    maxconn 51200
    chroot /usr/local/haproxy
    uid 0
    gid 0
    nbproc 1
    pidfile /var/run/haproxy.pid

defaults
    mode http
    option redispatch
    option abortonclose
    balance roundrobin
